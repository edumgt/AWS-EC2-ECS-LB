# AWS EC2 Auto - 점검용 조회 명령 모음

- 보안상 민감할 수 있는 리소스 ID는 마스킹 처리했습니다.

```
aws elbv2 describe-load-balancers --names <LB_NAME> --query "LoadBalancers[].AvailabilityZones[].SubnetId"
```
```
aws ec2 describe-subnets `
  --subnet-ids subnet-xxxxxxxx subnet-xxxxxxxx `
  --query "Subnets[].{SubnetId:SubnetId, AZ:AvailabilityZone}" `
  --output table
```
```
aws ec2 describe-route-tables `
  --filters Name=association.subnet-id,Values=subnet-xxxxxxxx,subnet-xxxxxxxx `
  --query "RouteTables[].Routes[?GatewayId!=null].GatewayId"
```
```
aws ec2 describe-security-groups `
  --group-ids sg-xxxxxxxx `
  --query "SecurityGroups[0].{GroupName:GroupName, GroupId:GroupId, VpcId:VpcId, Inbound:IpPermissions, Outbound:IpPermissionsEgress}" `
  --output json
```
